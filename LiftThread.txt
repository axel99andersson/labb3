package Threads;

import monitor.LiftMonitor;
import lift.*;

public class LiftThread extends Thread {
	private LiftView view;
	private LiftMonitor mon;
	
	public LiftThread(LiftView view, LiftMonitor mon) {
		this.view = view;
		this.mon = mon;
	}
	
	@Override
	public void run() {
		while(true) {
			try {
			int here = mon.getFloor();
			int dir = mon.getDir();
			mon.waitForPassenger();
			view.moveLift(here, here + dir);
			mon.setFloor(here + dir);
			
			if(mon.stopLift()) {
				view.openDoors(here + dir);
				mon.closeDoors();
				view.closeDoors();
			}
		
			} catch(InterruptedException e) {
				e.printStackTrace();
				throw new Error("Exception in Lift-thread", e);
			}
		}
	}
}
